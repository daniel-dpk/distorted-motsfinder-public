Detailed Description

Surface with S^2 topology immersed in space.

The surface is represented by three independent component functions and is hence not limited to be star-shaped.

Examples

This example will create a surface from three component functions. The surface is smooth (even at the coordinate poles) and hence will converge when represented in spherical harmonics.

%matplotlib inline
from math import sin, cos, exp
from motsfinder.ipyutils import plot_3d
from motsfinder.exprs import SphHarmSeries
from motsfinder.general3d.surface import ParametricSurface
# Define some arbitrary component functions
sph1 = SphHarmSeries([[0, 0, .8],
[0, 0, .2],
[0, -1, -.2],
[0, -.1, .2],
[0, 0, 1]]).evaluator()
def _to_xyz(uv):
from math import sin, cos
ta, ph = uv
return sin(ta)*cos(ph), sin(ta)*sin(ph), cos(ta)
def x_fun(uv):
x, y, z = _to_xyz(uv)
return (sin(uv[0]) * cos(uv[1])
+ 0.1*exp(x**2 - y**2 + sin(z))
- 0.05 * sin(x+y+z**2))
def y_fun(uv):
x, y, z = _to_xyz(uv)
return (sin(uv[0]) * sin(uv[1])
+ 0.1*exp(x**2 + y**2 - 0.5*sin(z))
+ 0.05 * sin(x**2+y+z))
def z_fun(uv):
return exp(sph1(uv)) * cos(uv[0])
# Create the surface.
surf = ParametricSurface.from_components(
x_fun, y_fun, z_fun, N=20,
)
# Plot it.
plot_3d(surf, lw=0.1, color='w')
